import java.util.Scanner;

public class filereader {
	
	//With assistance from this website: https://www.javatpoint.com/armstrong-number-in-java
	public static void main(String[] args)  {  
	    int c=0,a,temp;  
	    int n;
	   
	    System.out.println("Please enter your number");
	    Scanner input = new Scanner(System.in);
	    n = input.nextInt();
	    input.close();
	    
	    temp=n;  
	    while(n>0)  
	    {  
	    a=n%10;  
	    n=n/10;  
	    c=c+(a*a*a);  
	    }  
	    if(temp==c)  
	    System.out.println("IT IS AN armstrong number, YEEEAAAAAAAAAA!!!!!!!!!!!!!!!!!");   
	    else  
	        System.out.println("Not armstrong number. SHAME");   
	   }  

}
